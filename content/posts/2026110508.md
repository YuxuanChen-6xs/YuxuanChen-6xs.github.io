import hashlib, xml.etree.ElementTree as ET, time
from openai import OpenAI

app = Flask(__name__)

# 企业微信配置（替换成你自己的）
WX_APPID = "你的企业号CorpID"
WX_TOKEN = "你设置的回调Token"
WX_ENCODING_AES_KEY = "你设置的EncodingAESKey"

# 千聚API配置
client = OpenAI(
    base_url="https://www.qianjuai.com/v1",
    api_key="你千聚的API Key"
)

def verify_signature(signature, timestamp, nonce):
    """验证微信回调签名（简化版）"""
    arr = [WX_TOKEN, timestamp, nonce]
    arr.sort()
    return hashlib.sha1("".join(arr).encode()).hexdigest() == signature

@app.route("/callback", methods=["GET", "POST"])
def callback():
    # GET请求：企业微信验证服务器
    if request.method == "GET":
        signature = request.args.get("msg_signature")
        timestamp = request.args.get("timestamp")
        nonce = request.args.get("nonce")
        echostr = request.args.get("echostr")
        if verify_signature(signature, timestamp, nonce):
            return echostr
        return "fail"

    # POST请求：接收用户消息
    xml_data = request.data
    root = ET.fromstring(xml_data)
    content = root.find("Content").text if root.find("Content") is not None else ""
    from_user = root.find("FromUserName").text

    # 调用千聚AI生成回复
    response = client.chat.completions.create(
        model="gpt-4o",  # 可以换成 gpt-4o-mini 或 deepseek-chat 等
        messages=[
            {"role": "system", "content": "你是企业微信的智能客服，请用中文友好地回答用户问题。"},
            {"role": "user", "content": content}
        ],
        stream=False
    )
    reply = response.choices[0].message.content

    # 构造回复XML
    reply_xml = f"""<xml>
<ToUserName><![CDATA[{from_user}]]></ToUserName>
<FromUserName><![CDATA[{WX_APPID}]]></FromUserName>
<CreateTime>{int(time.time())}</CreateTime>
<MsgType><![CDATA[text]]></MsgType>
<Content><![CDATA[{reply}]]></Content>
</xml>"""
    return reply_xml, 200, {"Content-Type": "application/xml"}

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
