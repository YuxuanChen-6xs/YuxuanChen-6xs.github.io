## 核心脚本：Node.js 原生并发测试

这段脚本没有用任何第三方库，直接依赖 Node.js 内置的 `https` 模块，纯粹、轻量，方便你直接复制。

javascript
const https = require('https');

// 配置你的 API Key 和 中转站地址
const API_KEY = '你的千聚API密钥';
const BASE_URL = 'https://www.qianjuai.com/v1';

// 构造请求体
const requestBody = JSON.stringify({
  model: 'gpt-4.1-nano',
  messages: [{ role: 'user', content: '请用一句话介绍你自己。' }],
  max_tokens: 50,
  temperature: 0.7
});

// 并发参数
const CONCURRENCY = 300;
const results = { success: 0, fail: 0, errors: [] };
const startTime = Date.now();

// 发送单个请求的 Promise 包装
function sendRequest(index) {
  return new Promise((resolve) => {
    const url = new URL('/chat/completions', BASE_URL);

    const options = {
      hostname: url.hostname,
      path: url.pathname,
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Length': Buffer.byteLength(requestBody)
      }
    };

    const req = https.request(options, (res) => {
      let data = '';

      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        try {
          const parsed = JSON.parse(data);
          if (parsed.choices && parsed.choices[0]) {
            results.success++;
            resolve({ index, status: 'OK' });
          } else {
            results.fail++;
            results.errors.push({ index, error: parsed.error || '无choices' });
            resolve({ index, status: 'FAIL' });
          }
        } catch (e) {
          results.fail++;
          results.errors.push({ index, error: e.message });
          resolve({ index, status: 'FAIL' });
        }
      });
    });

    req.on('error', (e) => {
      results.fail++;
      results.errors.push({ index, error: e.message });
      resolve({ index, status: 'NETWORK_ERROR' });
    });

    req.write(requestBody);
    req.end();
  });
}

// 启动并发
(async () => {
  const promises = [];
  for (let i = 0; i < CONCURRENCY; i++) {
    promises.push(sendRequest(i));
  }

  console.log(`🚀 开始并发 ${CONCURRENCY} 次请求...`);
  await Promise.all(promises);
  const elapsed = Date.now() - startTime;

  console.log(`✅ 总耗时：${elapsed}ms`);
  console.log(`成功：${results.success} / 失败：${results.fail}`);
  if (results.errors.length > 0) {
    console.log('错误详情：');
    results.errors.forEach(e => console.log(`  #${e.index}: ${e.error}`));
  }
})();

注意这里有一个细节：单线程在 Node.js 里其实只有一个进程在处理所有请求的回调，但因为 HTTP 请求是 I/O 密集型操作，Node 的事件循环机制天然适合这种场景。300 个请求并行发出后，等待响应的过程不会阻塞主线程，所以实际吞吐量远高于顺序执行。
