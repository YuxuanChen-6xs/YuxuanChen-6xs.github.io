### Python 自动充值脚本示例

python

import requests
import time

# ----- 配置区域：填你自己的信息 -----
API_KEY = "sk-换成你在千聚申请的Key"
QUOTA_THRESHOLD = 10.0   # 余额低于多少美元时触发自动充值
TOP_UP_AMOUNT = 50.0     # 每次自动充多少钱（单位：美元）
BASE_URL = "https://www.qianjuai.com/v1"
# ----- 配置结束 -----

def check_balance():
    """查询当前余额"""
    headers = {"Authorization": f"Bearer {API_KEY}"}
    resp = requests.get(f"{BASE_URL}/dashboard/billing/credit_grants", headers=headers)
    data = resp.json()
    # 这里根据千聚实际返回的数据结构调整字段
    total_available = data.get("total_available", 0.0)
    return total_available

def auto_topup():
    """余额不足时自动充值"""
    balance = check_balance()
    print(f"当前余额：${balance:.2f}")

    if balance < QUOTA_THRESHOLD:
        print("余额低于阈值，触发自动充值...")
        payload = {
            "amount": TOP_UP_AMOUNT,
            "type": "balance"
        }
        headers = {
            "Authorization": f"Bearer {API_KEY}",
            "Content-Type": "application/json"
        }
        resp = requests.post(f"{BASE_URL}/dashboard/billing/topup", json=payload, headers=headers)
        if resp.status_code == 200:
            print(f"充值成功！已充值 ${TOP_UP_AMOUNT}")
        else:
            print(f"充值失败：{resp.text}")
    else:
        print("余额充足，无需充值。")

if __name__ == "__main__":
    # 循环检查：每10分钟检查一次余额
    while True:
        auto_topup()
        time.sleep(600)  # 600秒 = 10分钟
