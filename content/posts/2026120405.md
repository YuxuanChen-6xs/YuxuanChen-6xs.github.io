## 复杂业务场景：对话续传与上下文管理

如果只是单轮对话，那上面的代码就够了。但对于企业应用，比如客服系统或者文档助手，管理和跟踪会话状态才是痛点。下面是针对DeepSeek V3.2优化过的、带上下文管理的Node.js实现：

javascript
const OpenAI = require('openai');
const client = new OpenAI({
    baseURL: 'https://www.qianjuai.com/v1',
    apiKey: process.env.QIANJU_API_KEY,
});

// 模拟一个会话缓存
const conversationCache = new Map();

class DeepSeekSession {
    constructor(sessionId) {
        this.sessionId = sessionId;
        this.messages = [
            { role: 'system', content: '你是一个严谨的企业级AI助手，所有回复必须附带技术依据。' }
        ];
    }

    async chat(userInput) {
        this.messages.push({ role: 'user', content: userInput });

        const completion = await client.chat.completions.create({
            model: 'deepseek-v3.2',
            messages: this.messages,
            temperature: 0.3, // 企业场景下，低温度意味着更高的确定性
        });

        const reply = completion.choices[0].message.content;
        this.messages.push({ role: 'assistant', content: reply });
        
        // 企业级：自动截断历史（保留最近10轮对话）
        if (this.messages.length > 20) {
            this.messages = [
                this.messages[0], // 系统提示词必须保留
                ...this.messages.slice(-18)
            ];
        }
        return reply;
    }
}

// 使用示例

session.chat('帮我检查这段代码的内存泄漏风险: ').then(console.log);

### 错误重试机制
考虑到网络抖动，特别是当你的Node.js服务部署在国内机房，而模型是海外调用时，虽然是直连，但依然建议加入重试队列。下面是基于千聚API的指数退避重试代码片段：

javascript
async function callWithRetry(messages, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await client.chat.completions.create({
                model: 'deepseek-v3.2',
                messages: messages,
            });
        } catch (error) {
            if (i === maxRetries - 1) throw error;
            const delay = Math.pow(2, i) * 1000;
            console.log(`第${i+1}次失败，等待${delay}ms后重试...`);
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}
