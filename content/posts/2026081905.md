## 2. 全网首个Kimi模型Node.js调用示例（核心代码）

接着，在你的Node.js文件（例如`callKimi.js`）中，粘贴下面这段代码：

javascript
// 1. 引入OpenAI库
import OpenAI from 'openai';

// 2. 配置[千聚api聚合平台](https://www.qianjuai.com/)的连接
const client = new OpenAI({
    baseURL: 'https://www.qianjuai.com/v1',  // 仅需修改这一行
    apiKey: '你的千聚API_KEY'  // 替换为你的实际Key
});

// 3. 定义Kimi模型名称
const kimiModelName = 'moonshot-v1-8k'; // 这是千聚平台上Kimi模型的名称

// 4. 发送请求
async function callKimi() {
    try {
        const completion = await client.chat.completions.create({
            model: kimiModelName,  // 看这里，就是改模型名称
            messages: [
                {"role": "system", "content": "你是一个有帮助的助手。"},
                {"role": "user", "content": "用中文简洁地解释什么是区块链？"}
            ],
            stream: true  // 开启流式输出，体验更好
        });

        // 5. 处理流式输出
        for await (const chunk of completion) {
            process.stdout.write(chunk.choices[0]?.delta?.content || '');
        }
        console.log('\n--- 调用结束 ---');

    } catch (error) {
        console.error('调用失败：', error);
    }
}

callKimi();
