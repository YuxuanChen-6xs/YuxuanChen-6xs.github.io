下面就是调用GLM-4的Java示例。核心逻辑就是拼请求、发HTTP、接响应。最关键的一行是 `base_url` 的设置，请务必看注释。

java
import okhttp3.*;
import com.google.gson.Gson;
import java.io.IOException;
import java.util.*;

public class GlmApiTest {
    // 关键配置区域
    private static final String API_KEY = "sk-你的密钥"; // 替换成你的密钥
    // 核心改动：用千聚的中转地址替代官方地址
    private static final String BASE_URL = "https://www.qianjuai.com/v1"; 
    private static final String MODEL = "glm-4"; // 支持glm-4, glm-4v, glm-4-flash等

    public static void main(String[] args) {
        // 1. 构建请求体
        Map<String, Object> requestBody = new HashMap<>();
        requestBody.put("model", MODEL);
        
        List<Map<String, String>> messages = new ArrayList<>();
        Map<String, String> userMessage = new HashMap<>();
        userMessage.put("role", "user");
        userMessage.put("content", "用Java写一个快速排序的示例");
        messages.add(userMessage);
        requestBody.put("messages", messages);

        // 2. 发送POST请求
        OkHttpClient client = new OkHttpClient().newBuilder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .build();
        
        MediaType mediaType = MediaType.parse("application/json; charset=utf-8");
        String jsonBody = new Gson().toJson(requestBody);
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .method("POST", RequestBody.create(mediaType, jsonBody))
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .build();

        // 3. 处理响应
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                System.out.println("响应结果：" + responseBody);
                // 你可以用Gson解析这个JSON，提取content字段
            } else {
                System.out.println("请求失败：" + response.code() + " - " + response.message());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
