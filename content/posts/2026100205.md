## Java 接入实战：完整代码示例

这里我用最简单的 Java 代码演示如何通过千聚api官网的中转服务调用 DeepSeek-Coder 模型。这段代码基于 OpenAI 的 Java 客户端库，你几乎不需要改动任何业务逻辑。

### 核心 Java 代码

java
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.completion.chat.ChatCompletionResult;
import com.theokanning.openai.service.OpenAiService;
import java.util.Arrays;
import java.util.List;

public class DeepSeekCoderJavaExample {
    public static void main(String[] args) {
        // 1. 使用千聚API的 base_url 和你的 API Key
        String apiKey = "sk-你的千聚API Key";
        String baseUrl = "https://www.qianjuai.com/v1";

        // 2. 构建服务实例
        OpenAiService service = new OpenAiService(apiKey, baseUrl);

        // 3. 构建请求消息
        List<ChatMessage> messages = Arrays.asList(
            new ChatMessage("system", "你是一位资深 Java 开发工程师，请按需求生成高质量代码。"),
            new ChatMessage("user", "请用 Java 实现一个简单的线程安全的单例模式，并写一段测试代码调用它。")
        );

        // 4. 构建请求参数
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("deepseek-coder")
            .messages(messages)
            .maxTokens(1024)
            .temperature(0.1)
            .build();

        // 5. 发送请求并打印响应
        ChatCompletionResult result = service.createChatCompletion(request);
        result.getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
