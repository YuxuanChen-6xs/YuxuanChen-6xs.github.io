Java生态里调用OpenAI/Claude兼容的API，最常用的库是 `openai-java`（官方或社区版）。但在本例中，我们直接使用最底层的 `OkHttp` 和 `Gson` 来发送HTTP请求，这样最直观，你一看就懂。

java
import okhttp3.*;
import com.google.gson.Gson;
import com.google.gson.JsonObject;

import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class CallClaudeViaQianju {

    // 关键！这是千聚API的地址（兼容OpenAI格式）
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    // 替换成你的千聚API Key
    private static final String API_KEY = "sk-你的千聚API密钥";

    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(60, TimeUnit.SECONDS)
            .readTimeout(60, TimeUnit.SECONDS)
            .build();

    public static void main(String[] args) throws IOException {
        // 构建请求体（使用兼容格式调用Claude）
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "claude-3-5-sonnet-20240620"); // Claude 3.5 Sonnet
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "用一句诗形容编程的快乐。"); // 测试内容
        requestBody.add("messages", new Gson().toJsonTree(new JsonObject[]{message}));
        requestBody.addProperty("max_tokens", 200);
        requestBody.addProperty("stream", false);

        // 构建请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        new Gson().toJson(requestBody)
                ))
                .build();

        // 发送请求并获取响应
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.out.println("请求失败: " + response.code() + " " + response.message());
                return;
            }
            String responseBody = response.body().string();
            System.out.println("响应结果: " + responseBody);

            // 解析响应，提取AI回复内容
            JsonObject jsonResponse = new Gson().fromJson(responseBody, JsonObject.class);
            JsonObject choice = jsonResponse.getAsJsonArray("choices").get(0).getAsJsonObject();
            String content = choice.getAsJsonObject("message").get("content").getAsString();
            System.out.println("Claude的回答: " + content);
        }
    }
}
