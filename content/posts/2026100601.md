java
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;
import java.time.Duration;
import java.util.List;

public class ClaudeJavaDemo {
    public static void main(String[] args) {
        
        // [千聚ai聚合站](https://www.qianjuai.com/)的API密钥（需要替换成你自己的）
        String apiKey = "你的[千聚ai聚合站](https://www.qianjuai.com/)_sk-xxxx";
        
        // 关键步骤：修改为[千聚ai聚合站](https://www.qianjuai.com/)的API地址
        String baseUrl = "https://www.qianjuai.com/v1";
        
        // 创建OpenAiService实例
        OpenAiService service = new OpenAiService(
            apiKey, 
            Duration.ofSeconds(60)
        );
        // 注意：这里是自定义baseUrl的调用方式，不同客户端库写法可能不同
        // 也可以直接在service初始化时传入自定义的Retrofit配置

        // 构建请求消息
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("claude-3-5-sonnet-20241022") // 指定Claude模型
            .messages(List.of(
                new ChatMessage("user", "用Java写一个Hello World程序")
            ))
            .build();

        // 发送请求并获取回复
        service.createChatCompletion(request)
            .getChoices()
            .forEach(choice -> {
                System.out.println("Claude回答： " + choice.getMessage().getContent());
            });
    }
}


代码逻辑简单到不需要注释说明。唯一需要你动手改的，就是在调用前把 `base_url` 换成 `https://www.qianjuai.com/v1`。
