### 第一步：初始化项目并安装依赖

在电脑上找一个合适的地方，新建一个文件夹，例如 `gpt4o-node-demo`。打开终端或命令行（VSCode里内部终端也行），先进入该文件夹，然后依次输入以下命令：

bash
# 初始化一个node项目（一路回车即可）

# 安装OpenAI官方SDK
npm install openai


安装完成后，你的项目里会多出一个 `node_modules` 文件夹和 `package-lock.json` 文件。这时候核心依赖已经就位了。

### 第二步：创建主程序文件

在项目文件夹根目录下，新建一个 `index.js` 文件。现在我们把所有代码写在这个文件里。这里是实现“与GPT-4o对话”的完整代码示例：

javascript
// 第一步：引入OpenAI SDK
const OpenAI = require('openai');

// 第二步：初始化OpenAI客户端
const client = new OpenAI({
    // 🌟 重点1：替换base_url
    baseURL: 'https://www.qianjuai.com/v1',
    // 🌟 重点2：替换apiKey，请自行换成你在千聚后台生成的Key
    apiKey: 'sk-这里是你的千聚API Key',
});

// 第三步：异步调用函数，发起对话请求
async function main() {
    try {
        // 构建对话请求
        const completion = await client.chat.completions.create({
            // 定义调用的模型名：新手直接用GPT-4o最新版
            model: 'gpt-4o',
            // 定义消息历史，这里只发一条系统+用户消息就行
            messages: [
                { role: 'system', content: '你是一个友善且乐于助人的AI助手，精通全栈开发。' },
                { role: 'user', content: '请用中国谚语“授人以鱼不如授人以渔”的解释，给我举一个前后端分离设计的Node.js例子。' },
            ],
            // 可选：规定最大token输出量
            max_tokens: 1024,
            // 可选：控制创造性的随机性
            temperature: 0.5,
        });

        // 打印返回结果
        console.log('GPT-4o返回的内容如下：');
        // 返回的消息在 choices[0].message.content
        console.log(completion.choices[0].message.content);

    } catch (error) {
        // 如果调用失败，打印错误信息（比如Key不对、网络不通等）
        console.error('糟糕，请求报错了，检查一下Key或网络吧: ', error.message);
    }
}

// 第四步：运行函数
main();
