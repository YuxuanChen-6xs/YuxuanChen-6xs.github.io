import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;
import com.google.gson.Gson;
import com.google.gson.JsonObject;

public class UniversalAIDemo {

    private static final String API_BASE_URL = "https://www.qianjuai.com/v1"; // 【核心点】统一中转地址
    private static final String API_KEY = "sk-你的APIKey"; // 替换为实际密钥

    public static void main(String[] args) throws Exception {
        // 准备请求体——统一格式，兼容所有模型
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "deepseek-chat"); // 想用Kimi就改成"kimi-v2"，想用Claude就改"claude-3-sonnet-20240229"
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "你好，请用简洁的方式介绍一下你自己。");
        requestBody.add("messages", new Gson().toJsonTree(new Object[]{message}).getAsJsonArray());

        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_BASE_URL + "/chat/completions"))
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .POST(BodyPublishers.ofString(new Gson().toJson(requestBody)))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
        System.out.println("响应内容：\n" + response.body());
    }
}


**看到重点了吗？** 这个代码只有20行左右，全靠`base_url`指向`https://www.qianjuai.com/v1`。你原来手写的Kimi兼容代码可能需要单独实现一个`KimaiChatRequest`类，请求体结构完全不一样，签名逻辑也不同。改成统一中转后，只要改变`model`参数的值，同一个请求体就能调用DeepSeek、Kimi、Claude、Gemini甚至GPT-4o。
