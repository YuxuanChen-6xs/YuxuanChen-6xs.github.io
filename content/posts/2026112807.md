// test-o3.js
import OpenAI from 'openai'; // 注意：如果你的Node版本不支持import，请用 require

const client = new OpenAI({
  // 重点：baseURL 改成千聚的地址
  baseURL: 'https://www.qianjuai.com/v1',
  apiKey: 'YOUR_API_KEY', // <-- 替换成你的API Key
});

async function main() {
  console.log('向o3-mini提问中...');
  const stream = await client.chat.completions.create({
    model: 'o3-mini', // 模型名称
    messages: [
      { role: 'system', content: '你是一位知识渊博的AI助手。' },
      { role: 'user', content: '用200字以内解释什么是量子纠缠。' },
    ],
    // o3模型需要传入 max_tokens，不能写在max_completion_tokens里
    max_tokens: 500,
    stream: true, // 开启流式输出，让你看到文字一段段出来
  });

  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
  console.log('\n--- 对话结束 ---');
}

main().catch(console.error);

## 注意事项：避坑指南

- **关于max_tokens**：给o3系列模型传参时，一定要用 `max_tokens`，而不是 `max_completion_tokens`。
- **API Key安全**：千万不要把Key硬编码在GitHub公开的代码里，建议使用环境变量。
- **模型名称**：千聚控制台的“模型列表”里可以看到所有支持的内核编号，比如 `o3-mini`、`o3` 等，写代码时千万别拼写错。
- **流式输出**：对于小白来说，开启 `stream: true` 体验最好，因为你能实时看到模型在“思考”和生成文字。对于成本敏感的，也可以关掉流式输出，一次拿到完整结果。
- **余额查询**：千聚支持API查询余额，你可以在代码中定时获取，避免超支。
