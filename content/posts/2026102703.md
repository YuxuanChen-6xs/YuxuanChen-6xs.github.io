import okhttp3.*;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

public class QGPT5MiniDemo {
    
    private static final String API_KEY = "你的千聚API密钥"; // 替换成你在www.qianjuai.com申请的key
    private static final String BASE_URL = "https://www.qianjuai.com/v1"; // 关键替换点
    private static final MediaType JSON = MediaType.get("application/json; charset=utf-8");

    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient();
        
        // 构建请求体，指定模型为GPT-5-mini（或支持的官方名称）
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "gpt-5-mini"); 
        JsonObject systemMessage = new JsonObject();
        systemMessage.addProperty("role", "system");
        systemMessage.addProperty("content", "你是一个专业的AI助手。");
        JsonObject userMessage = new JsonObject();
        userMessage.addProperty("role", "user");
        userMessage.addProperty("content", "请用Java写一个排序算法。");
        
        // 构建消息数组
        var messages = new com.google.gson.JsonArray();
        messages.add(systemMessage);
        messages.add(userMessage);
        requestBody.add("messages", messages);
        
        // 设置流式输出（可优化成本）
        requestBody.addProperty("stream", true);
        
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(requestBody.toString(), JSON))
                .build();
        
        // 处理流式响应
        client.newCall(request).enqueue(new Callback() {
            @Override
            public void onFailure(Call call, IOException e) {
                System.err.println("请求失败: " + e.getMessage());
            }

            @Override
            public void onResponse(Call call, Response response) throws IOException {
                if (!response.isSuccessful()) {
                    System.err.println("服务器错误: " + response.code());
                    return;
                }
                // 处理流式输出，逐块打印
                try (ResponseBody body = response.body()) {
                    if (body != null) {
                        // 使用OkHttp的流处理，模拟逐chunk输出
                        System.out.println("开始接收流式数据...");
                        // 注意：这里简化了流处理逻辑，生产环境建议使用SSE解析器
                    }
                }
            }
        });
    }
}


**代码核心逻辑：**
1.  **替换核心地址**：`BASE_URL` 从 `https://api.openai.com/v1` 改成 `https://www.qianjuai.com/v1`。
2.  **使用正确的模型名**：`model` 字段用 `gpt-5-mini`（或平台支持的官方ID）。
3.  **开启流式输出**：设 `stream=true`。这不仅是实现更好的用户体验，更是省钱诀窍——流式输出能大幅减少显存占用，降低单次请求的Token消耗。
4.  **成本杀手**：如果你做的是客服问答、知识库检索这类应用，强烈建议接入千聚的**缓存功能**。实现方式：在请求头添加 `X-QJ-Cache: true` 即可让系统缓存相似问题的回复，下次同样的Token请求直接命中免费缓存，费用直接变为0。
