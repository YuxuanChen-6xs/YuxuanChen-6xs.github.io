### 完整Java调用示例
java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.*;

import java.util.Arrays;
import java.util.List;

public class Qwen3CoderDemo {
    public static void main(String[] args) {
        // 1. 配置千聚API地址
        String token = "你的千聚API Key"; // 在千聚后台申请
        OpenAiService service = new OpenAiService(
            token,
            Duration.ofSeconds(60)
        );
        
        // 2. 构造消息
        ChatMessage systemMessage = new ChatMessage("system", "你是一位资深Java架构师，擅长代码生成与重构。");
        ChatMessage userMessage = new ChatMessage("user", "请用Java写一个高性能的线程池工具类，支持动态调整核心线程数、最大线程数和队列大小。");
        
        List<ChatMessage> messages = Arrays.asList(systemMessage, userMessage);
        
        // 3. 调用模型
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("gpt-4o-mini") // 千聚自动映射到qwen3-coder
            .messages(messages)
            .maxTokens(4096)
            .temperature(0.7)
            .build();
        
        ChatCompletionResult result = service.createChatCompletion(request);
        System.out.println(result.getChoices().get(0).getMessage().getContent());
    }
}
