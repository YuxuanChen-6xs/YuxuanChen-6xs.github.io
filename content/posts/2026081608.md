**终极稳定版：纯Java + OkHttp3 调用Qwen模型**

java
import okhttp3.*;
import org.json.JSONArray;
import org.json.JSONObject;
import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class QwenOkHttpDemo {

    public static void main(String[] args) {
        // 1. 你的API密钥
        String apiKey = "sk-你的千聚API_Key"; 
        // 2. 千聚API中转站地址 (核心)
        String baseUrl = "https://www.qianjuai.com/v1";

        // 3. 构建请求体
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "qwen-turbo"); // Qwen的模型名，你可以在后台看到完整的

        JSONArray messages = new JSONArray();
        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");
        userMessage.put("content", "用Java写一个简单的Hello World程序，只需要核心代码。");
        messages.put(userMessage);
        requestBody.put("messages", messages);

        // 4. 发起HTTP请求
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .build();

        MediaType JSON = MediaType.parse("application/json; charset=utf-8");
        Request request = new Request.Builder()
                .url(baseUrl + "/chat/completions") // 直接把baseUrl拼上路径
                .addHeader("Authorization", "Bearer " + apiKey)
                .post(RequestBody.create(JSON, requestBody.toString()))
                .build();

        try {
            Response response = client.newCall(request).execute();
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                JSONObject jsonObject = new JSONObject(responseBody);
                String result = jsonObject.getJSONArray("choices")
                        .getJSONObject(0)
                        .getJSONObject("message")
                        .getString("content");
                System.out.println("Qwen机器人回复: ");
                System.out.println(result);
            } else {
                System.out.println("请求失败: " + response.code() + " " + response.message());
                System.out.println("错误详情: " + response.body().string());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
