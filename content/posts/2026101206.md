## 第三步：写一段代码，调用DeepSeek R1

现在就是见证奇迹的时刻。

在你的项目文件夹里，新建一个`index.js`文件，然后粘贴下面这段代码。代码里我写好了注释，一看就懂。

javascript
// index.js
// 1. 引入 OpenAI 库
const OpenAI = require('openai');

// 2. 创建 OpenAI 客户端实例
const client = new OpenAI({
    // 关键1：替换 base_url 为千聚的接口地址
    baseURL: 'https://www.qianjuai.com/v1',
    // 关键2：把你的 API Key 粘贴在这里（请替换成你自己的）
    apiKey: 'sk-你的千聚APIKey', 
});

// 3. 定义一个异步函数，用于和 DeepSeek R1 对话
async function chatWithDeepSeek() {
    try {
        const response = await client.chat.completions.create({
            // 这里指定调用 DeepSeek R1 满血版
            model: 'deepseek-r1-250120', 
            messages: [
                {'role': 'user', 'content': '请用一句话解释什么是量子纠缠？'},
            ],
            // 开启流式输出，会像网页对话一样逐字展示回复
            stream: true,
        });

        // 4. 处理流式返回的数据
        for await (const chunk of response) {
            process.stdout.write(chunk.choices[0]?.delta?.content || '');
        }
    } catch (error) {
        console.error('API调用失败：', error.response ? error.response.data : error.message);
    }
}

// 5. 执行对话
chatWithDeepSeek();


**代码核心要点：**
*   `baseURL`：必须改成千聚的API域名 `https://www.qianjuai.com/v1`，这是接入的中转关键。
*   `model`：用 `deepseek-r1-250120`，这是千聚支持的DeepSeek R1模型名称。
*   `stream: true`：开启流式输出，体验更好，就像在官网聊天一样。

## 进阶：构建一个命令行对话机器人

如果你觉得上面单次调用不过瘾，这里还准备了一个简单的命令行对话机器人代码，让你可以连续跟DeepSeek R1聊天。新建一个`chatbot.js`文件，复制下面代码：

javascript
const OpenAI = require('openai');
const readline = require('readline').createInterface({
    input: process.stdin,
    output: process.stdout,
});

const client = new OpenAI({
    baseURL: 'https://www.qianjuai.com/v1',
    apiKey: 'sk-你的千聚APIKey', // 千万别忘了替换
});

// 维护一个对话历史
let history = [];

async function askDeepSeek(userMessage) {
    history.push({ role: 'user', content: userMessage });

    const response = await client.chat.completions.create({
        model: 'deepseek-r1-250120',
        messages: history,
        stream: true,
    });

    let fullContent = '';
    process.stdout.write('DeepSeek: ');

    for await (const chunk of response) {
        const content = chunk.choices[0]?.delta?.content || '';
        process.stdout.write(content);
        fullContent += content;
    }

    process.stdout.write('\n');
    history.push({ role: 'assistant', content: fullContent });
    startChat();
}

function startChat() {
    readline.question('你: ', async (input) => {
        if (input.toLowerCase() === 'exit' || input.toLowerCase() === '退出') {
            console.log('再见！');
            readline.close();
            process.exit(0);
        }
        await askDeepSeek(input);
    });
}

console.log('开始和 DeepSeek R1 对话吧！输入"exit"退出聊天。');
startChat();


运行 `node chatbot.js`，你就可以体验连续对话了。这个示例展示了如何管理对话上下文、处理流式输出和异常。
