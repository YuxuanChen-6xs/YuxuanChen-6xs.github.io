## 第二步：用 Node.js 调用通义千问（完整代码）

我们先写一个最简单的示例：让通义千问帮你写一首诗。

在项目根目录下创建一个 `index.js` 文件，内容如下：

javascript
const OpenAI = require('openai');

// 创建 OpenAI 客户端实例
const client = new OpenAI({
  // 关键：把 baseURL 换成千聚的接口地址
  baseURL: 'https://www.qianjuai.com/v1',
  // 换成你自己的 API Key
  apiKey: 'sk-你的apikey',
});

async function main() {
  try {
    const completion = await client.chat.completions.create({
      // 这里指定模型，通义千问最新版
      model: 'qwen-max',
      messages: [
        { role: 'system', content: '你是一个充满诗意的助手。' },
        { role: 'user', content: '请以“春天”为主题，写一首五言绝句。' },
      ],
      max_tokens: 200,
    });

    console.log('通义千问的回答：');
    console.log(completion.choices[0]?.message?.content);
  } catch (error) {
    console.error('调用出错：', error);
  }
}

main();


然后运行它：
