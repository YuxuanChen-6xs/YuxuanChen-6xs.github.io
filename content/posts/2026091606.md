import org.springframework.http.*;
import org.springframework.web.client.RestTemplate;

public class DoubaoApiExample {
    public static void main(String[] args) {
        // 关键改动：把基地址从官方地址改成[千聚ai聚合站](https://www.qianjuai.com/)的
        String baseUrl = "https://www.qianjuai.com/v1";
        String apiKey = "你的千聚API-KEY";

        // 请求体构造（OpenAI 标准格式，兼容豆包模型）
        String requestJson = """
        {
            "model": "doubao-pro-32k",
            "messages": [
                {"role": "user", "content": "用专业术语解释什么是向量数据库"}
            ],
            "temperature": 0.7
        }
        """;

        RestTemplate client = new RestTemplate();
        HttpHeaders headers = new HttpHeaders();
        headers.setBearerAuth(apiKey);
        headers.setContentType(MediaType.APPLICATION_JSON);

        HttpEntity<String> entity = new HttpEntity<>(requestJson, headers);
        ResponseEntity<String> response = client.postForEntity(
                baseUrl + "/chat/completions", entity, String.class);

        System.out.println(response.getBody());
    }
}
