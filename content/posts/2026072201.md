// app.js
import OpenAI from 'openai';
import dotenv from 'dotenv';

dotenv.config();

// 关键：所有模型走的都是千聚的同一个 base_url 和同一个 client
const client = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1', // 你的千聚 API 端点
  apiKey: process.env.QIANJU_API_KEY,      // 你的千聚 API Key
});

// 定义“模型品牌”与千聚平台模型 ID 的映射
const MODELS = {
  openai: 'gpt-4o',           // OpenAI 的最新旗舰
  claude: 'claude-3-5-sonnet-20240620',  // Claude 3.5 Sonnet (需官转分组支持)
  grok: 'grok-2-latest',      // Grok 最新版
};

// 通用 AI 调用函数：切换模型只需改一个参数
async function askAI(platform = 'openai', userMessage = '你好，请用一句话介绍自己。') {
  try {
    const modelId = MODELS[platform];
    if (!modelId) throw new Error(`不支持的平台：${platform}。可用：${Object.keys(MODELS).join(', ')}`);

    const response = await client.chat.completions.create({
      model: modelId,
      messages: [
        { role: 'system', content: '你是一个乐于助人的助手。' },
        { role: 'user', content: userMessage },
      ],
      max_tokens: 200,
      stream: false,
    });

    return response.choices[0].message.content;
  } catch (error) {
    console.error(`[${platform}] 调用失败：`, error.message);
    throw error;
  }
}

// 示例：连续调用三个模型
async function runDemo() {
  console.log('=== [千聚api聚合站](https://www.qianjuai.com/) · 三平台聚合演示 ===\n');

  // 1. 调用 OpenAI GPT-4o
  const openaiReply = await askAI('openai', '今天天气好吗？');
  console.log(`🤖 OpenAI回复：${openaiReply}\n`);

  // 2. 调用 Anthropic Claude 3.5
  const claudeReply = await askAI('claude', '写一句诗，关于深秋的夜晚。');
  console.log(`🧠 Claude回复：${claudeReply}\n`);

  // 3. 调用 X.AI Grok-2
  const grokReply = await askAI('grok', '用三个词形容你的性格。');
  console.log(`🚀 Grok回复：${grokReply}\n`);

  console.log('=== 演示结束，三个模型都接入完毕！ ===');
}

runDemo().catch(console.error);
