**步骤三：运行示例代码**

下面是完整的Java示例，调用Qwen-VL进行图片内容分析。这个代码可以运行在任何国内服务器上，不需要任何代理：

java
import okhttp3.*;
import org.json.JSONObject;

public class QwenVLExample {
    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient().newBuilder()
                .connectTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 1. 构建请求体
        JSONObject base = new JSONObject();
        base.put("model", "qwen-vl-plus"); // 千聚支持Qwen-VL全线模型

        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");

        JSONObject content = new JSONObject();
        content.put("type", "text");
        content.put("text", "请详细描述这张图片的内容，包括文字和布局。");

        JSONObject imageContent = new JSONObject();
        imageContent.put("type", "image_url");
        imageContent.put("image_url", new JSONObject().put("url", "https://www.your-domain.com/invoice.jpg"));

        userMessage.put("content", new Object[]{content, imageContent});

        base.put("messages", new JSONObject[]{userMessage});
        base.put("max_tokens", 2048);

        // 2. 发送请求（重点：base_url改成了千聚的地址）
        RequestBody body = RequestBody.create(
                MediaType.parse("application/json"),
                base.toString()
        );

        Request request = new Request.Builder()
                .url("https://www.qianjuai.com/v1/chat/completions")
                .addHeader("Authorization", "Bearer 你的千聚API Key")
                .post(body)
                .build();

        // 3. 解析响应
        Response response = client.newCall(request).execute();
        if (response.isSuccessful()) {
            String rawResponse = response.body().string();
            JSONObject result = new JSONObject(rawResponse);
            String aiAnswer = result.getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");
            System.out.println("Qwen-VL回答: " + aiAnswer);
        } else {
            System.out.println("请求失败，状态码: " + response.code());
        }
    }
}
