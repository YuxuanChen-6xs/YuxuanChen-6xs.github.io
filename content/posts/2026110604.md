### 第三步：编写核心调用代码（附完整的JSON解析）

java
import okhttp3.*;
import com.google.gson.*;
import java.io.IOException;

public class GlmApiClient {

    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(60, TimeUnit.SECONDS) // 关键：大模型流式响应超时设长一些
            .build();
    private static final Gson gson = new Gson();

    public static String callGlm(String apiKey, String baseUrl, String model, String userMessage) throws IOException {
        // 构建请求体（兼容OpenAI格式，千聚api无差别支持）
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", model);
        JsonArray messages = new JsonArray();
        JsonObject systemMessage = new JsonObject();
        systemMessage.addProperty("role", "system");
        systemMessage.addProperty("content", "你是一个有帮助的助手，请用中文回答。");
        messages.add(systemMessage);
        JsonObject userMsg = new JsonObject();
        userMsg.addProperty("role", "user");
        userMsg.addProperty("content", userMessage);
        messages.add(userMsg);
        requestBody.add("messages", messages);
        requestBody.addProperty("temperature", 0.7);
        
        // 发送POST请求
        Request request = new Request.Builder()
                .url(baseUrl + "/chat/completions") // 标准OpenAI格式接口
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(requestBody.toString(), MediaType.parse("application/json")))
                .build();
        
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("请求失败，HTTP状态码：" + response.code() + "，响应：" + response.body().string());
            }
            String responseBody = response.body().string();
            // 解析返回JSON，提取content
            JsonObject jsonObj = gson.fromJson(responseBody, JsonObject.class);
            String content = jsonObj.getAsJsonArray("choices")
                    .get(0).getAsJsonObject()
                    .get("message").getAsJsonObject()
                    .get("content").getAsString();
            return content;
        }
    }
    
    public static void main(String[] args) throws IOException {
        // 测试：使用你的千聚api key和base-url
        String apiKey = "sk-your-key"; // 替换为实际key
        String baseUrl = "https://www.qianjuai.com/v1";
        String model = "glm-4";
        String result = callGlm(apiKey, baseUrl, model, "请用中文介绍GLM-4模型的特点");
        System.out.println("GLM-4返回结果：" + result);
    }
}
