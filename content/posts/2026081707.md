### 第二步：粘贴以下代码

javascript
// test-grok.js
// 你不需要安装任何 npm 包，Node 18+ 原生支持 fetch

const API_KEY = 'YOUR_API_KEY'; // 替换成你刚复制的 Key
const BASE_URL = 'https://www.qianjuai.com/v1'; // [千聚ai中转站](https://www.qianjuai.com/) API 地址
const MODEL = 'grok-3'; // 或 'grok-3-latest'，以千聚模型列表为准

async function callGrok3() {
  try {
    const response = await fetch(`${BASE_URL}/chat/completions`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${API_KEY}`
      },
      body: JSON.stringify({
        model: MODEL,
        messages: [
          {
            role: 'user',
            content: '你好，请用中文简单介绍一下你自己，并说说你擅长什么。'
          // 如果是 Grok 3，可能会回答关于 xAI 的通用知识
        }
      ],
      max_tokens: 300, // 控制回复长度
      temperature: 0.7 // 控制创造性，0-2，推荐 0.7
    });

    if (!response.ok) {
      throw new Error(`HTTP 错误！状态：${response.status}，信息：${await response.text()}`);
    }

    const data = await response.json();
    console.log('Grok 3 回复：');
    // 提取大模型回复内容
    const reply = data.choices[0]?.message?.content || '无回复';
    console.log(reply);
  } catch (error) {
    console.error('调用失败：', error.message);
  }
}

callGrok3();

## 常见问题 & 小技巧

### 1. 一定要把 Node.js 才能跑吗？

不是。如果你没有 Node.js 环境，或者不想装，可以用任何一个支持 HTTP POST 的工具。

**用 Python 也可以**：把上面的 JavaScript 代码换成 Python 的 `requests` 库，一样简单：

python
import requests

url = "https://www.qianjuai.com/v1/char/completions"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}
data = {
    "model": "grok-3",
    "messages": [{"role": "user", "content": "你好}]
}
resp = requests.post(url, headers=headers, json=data)
print(resp.json()['choices'][0]['message']['content'])


**用在线平台**：比如 `https://hoo.io`（Hoppman）或或者 Postman，填入以上参数，也能瞬间调通。
