## 半小时搞定的Java接入示例

空谈无益，直接上代码。下面是一个极其简单的Java示例，使用原生的jdK http库来调用[千聚ai聚合站](https://www.qianjuai.com/)转发的Gemini API。完全不依赖任何第三方框架，0基础也能看懂。

**第一步：引依赖（可选）**

如果你用的是Maven，可以加入一个JSON处理库（方便解析响应）。甚至你也可以不用，自己手写一个JSON解析器。这里用阿里云的fastjson2，非常轻量，也适合新手。

**第二步：写一个简单的API调用类**

java
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;
import com.alibaba.fastjson2.JSONObject;

public class GeminiCaller {

    // 替换成你在千聚申请的API Key
    private static final String API_KEY = "your_apikey_from_qianjuai";
    // 千聚的API网关地址，只需替换base_url
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";

    public static String callGemini(String userMessage) throws Exception {
        // 构建请求体（兼容OpenAI格式）
        JSONObject payload = new JSONObject();
        payload.put("model", "gemini-2.0-flash"); // 指定Gemini模型
        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", userMessage);
        payload.put("messages", new JSONObject[]{message});
        // 设置温度等参数
        payload.put("temperature", 0.7);

        // 发送HTTP POST请求
        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL))
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .POST(BodyPublishers.ofString(payload.toJSONString()))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());

        // 解析响应
        if (response.statusCode() == 200) {
            JSONObject responseJson = JSONObject.parseObject(response.body());
            // 提取回复内容
            String reply = responseJson.getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");
            return reply;
        } else {
            throw new RuntimeException("API调用失败，状态码：" + response.statusCode() + "，响应：" + response.body());
        }
    }

    public static void main(String[] args) throws Exception {
        String prompt = "用Java写一段冒泡排序的代码，并解释其原理。";
        String result = callGemini(prompt);
        System.out.println("Gemini回复：\n" + result);
    }
}
