### 第二步：核心调用代码（零报错版本）

直接复制这段 `Service` 层代码，改掉你的 `api_key` 即可运行。我把每个关键点都做了注释。

java
import okhttp3.*;
import com.alibaba.fastjson2.JSON;
import com.alibaba.fastjson2.JSONObject;
import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class QianJuO4MiniService {
    
    // 你的API Key，去千聚ai注册后立刻就能拿到
    private static final String API_KEY = "your-api-key";
    // 核心URL：千聚API的完整接口地址
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    // 要调用的模型名称
    private static final String MODEL = "o4-mini";
    
    private final OkHttpClient client;
    
    public QianJuO4MiniService() {
        this.client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(120, TimeUnit.SECONDS)
                .writeTimeout(60, TimeUnit.SECONDS)
                .build();
    }
    
    /**
     * 发送对话消息到o4-mini模型
     * @param userMessage  用户输入的文字
     * @return 模型的回复内容
     */
    public String sendMessage(String userMessage) throws IOException {
        // 构建请求体
        JSONObject requestBody = new JSONObject()
                .fluentPut("model", MODEL)
                .fluentPut("messages", new JSONObject[]{
                        new JSONObject()
                                .fluentPut("role", "system")
                                .fluentPut("content", "你是顶级的AI助手。请用中文简洁、准确地回答用户问题。"),
                        new JSONObject()
                                .fluentPut("role", "user")
                                .fluentPut("content", userMessage)
                })
                .fluentPut("temperature", 0.7)  // 控制创造力，0到2
                .fluentPut("max_tokens", 4096);  // 单次回复的最大长度
        
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toJSONString()
                ))
                .build();
        
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("API调用失败，状态码: " + response.code() + 
                        "，错误信息: " + response.body().string());
            }
            String resultString = response.body().string();
            JSONObject resultJson = JSON.parseObject(resultString);
            // 解析出AI的回答内容
            return resultJson
                    .getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");
        }
    }
}
