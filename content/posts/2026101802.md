## Java 接入 o4-mini 实战——就是这么简单

很多开发者用 Java 做后端，现在千聚让 Java 接入 AI 模型变得异常简单。下面是一个完整的 Java 示例，演示如何用普通 HTTP 客户端调用 o4-mini 模型。

### 准备工作

1. 注册千聚API中转站账号，获取 API Key。
2. 确认你的 Java 版本支持 HttpURLConnection 或 Apache HttpClient（JDK 8+ 即可）。
3. 不需要任何第三方 AI SDK，只靠标准库。

### 代码示例

java
import java.io.*;
import java.net.*;
import java.nio.charset.StandardCharsets;

public class QianjuO4MiniExample {
    public static void main(String[] args) throws IOException {
        // 官方直充 -> 千聚中转站：只改 base_url 和 api_key
        String baseUrl = "https://www.qianjuai.com/v1"; // 关键行：改成千聚的接口地址
        String apiKey = "sk-你的千聚API密钥"; // 替换为你的真实密钥

        // 构建请求体
        String requestBody = "{"
                + "\"model\": \"o4-mini\","
                + "\"messages\": [{\"role\": \"user\", \"content\": \"用Java写一个快速排序算法\"}],"
                + "\"temperature\": 0.7"
                + "}";

        // 创建连接
        URL url = new URL(baseUrl + "/chat/completions");
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();
        conn.setRequestMethod("POST");
        conn.setDoOutput(true);
        conn.setRequestProperty("Content-Type", "application/json");
        conn.setRequestProperty("Authorization", "Bearer " + apiKey);

        // 发送请求
        try (OutputStream os = conn.getOutputStream()) {
            byte[] input = requestBody.getBytes(StandardCharsets.UTF_8);
            os.write(input, 0, input.length);
        }

        // 读取响应
        int status = conn.getResponseCode();
        if (status == HttpURLConnection.HTTP_OK) {
            BufferedReader reader = new BufferedReader(
                new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8)
            );
            StringBuilder response = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
            reader.close();
            System.out.println("o4-mini 响应：");
            System.out.println(response.toString());
        } else {
            System.out.println("请求失败，状态码：" + status);
            BufferedReader errorReader = new BufferedReader(
                new InputStreamReader(conn.getErrorStream(), StandardCharsets.UTF_8)
            );
            String errorLine;
            while ((errorLine = errorReader.readLine()) != null) {
                System.out.println(errorLine);
            }
            errorReader.close();
        }

        conn.disconnect();
    }
}
